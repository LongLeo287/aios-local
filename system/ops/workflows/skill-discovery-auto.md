## Phase 1 â€” Detect Missing SKILL.md (Auto-Scan)

```powershell
# Run at every corp boot (corp-daily-cycle Phase 0)
$ROOT = "<AI_OS_ROOT>"
$missing = @()

# Scan all relevant locations
foreach ($dir in @("plugins","tools","workforce/agents","subagents","system\ops\scripts")) {
  Get-ChildItem "$ROOT\$dir" -Directory | Where-Object {
    -not (Test-Path "$($_.FullName)\SKILL.md")
  } | ForEach-Object {
    $missing += @{ path = $_.FullName; name = $_.Name; dir = $dir }
  }
  
  # Äáº·c biá»‡t Æ°u tiÃªn file script Python/PS1 Ä‘á»™c láº­p:
  if ($dir -eq "system\ops\scripts") {
    Get-ChildItem "$ROOT\$dir" -File -Include *.py,*.ps1 | Where-Object {
       -not (Test-Path "$ROOT\ecosystem\skills\$($_.BaseName)\SKILL.md")
    } | ForEach-Object {
       $missing += @{ path = $_.FullName; name = $_.BaseName; dir = "scripts (standalone)" }
    }
  }
}

if ($missing.Count -gt 0) {
  Write-Host "[REGISTRY] $($missing.Count) locations need SKILL.md"
  # Write to blackboard for registry-manager-agent to pick up
  # blackboard.pending_skills = $missing
  $missing | ForEach-Object { Write-Host "  - $($_.dir)/$($_.name)" }
} else {
  Write-Host "[REGISTRY] All locations have SKILL.md. No action needed."
}
```
